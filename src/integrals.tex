\chapter{Pochodne}

\begin{proposition}
    \label{prp:derivative_power}%
    \begin{equation}
        \frac{\mathrm{d}}{\mathrm{d}x} x^n = nx^{n-1}.
    \end{equation}
\end{proposition}

\chapter{Całki}

\begin{integral}
    Niech $n \neq -1$.
    Wtedy
\begin{equation}
    \int x^n \,\mathrm{d}x = \frac{x^{n+1}}{n+1}.
\end{equation}
\end{integral}

\begin{proof}
    Wprost z \ref{prp:derivative_power}.
\end{proof}
